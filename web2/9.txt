    <script type="text/javascript">
      function validate() {
        var userId = document.getElementById("user").value;
        var name = document.getElementById("name").value;
        var password = document.getElementById("password").value;
        var email = document.getElementById("email").value;

        var userIdRegex = /^[a-zA-Z0-9]{5,12}$/;
        var nameRegex = /^[a-zA-Z ]{15,}$/;
        var passwordRegex = /^(?=.*[a-z])(?=.*[A-Z])(?=.*\\d)(?=.*[@$!%*?&])[A-Za-z\\d@$!%*?&]{8,}$/;
        var emailRegex = /^[^\\s@]+@[^\\s@]+\\.[^\\s@]+$/;

        if (!userIdRegex.test(userId)) {
          alert("User ID must be between 5 to 12 characters and alphanumeric.");
          return false;
        }

        if (!nameRegex.test(name)) {
          alert("Name must be at least 15 alphabetic characters.");
          return false;
        }

        if (!passwordRegex.test(password)) {
          alert("Password must be 8+ characters, one uppercase, one symbol, and numbers.");
          return false;
        }

        if (!emailRegex.test(email)) {
          alert("Invalid email format.");
          return false;
        }

        alert("Form submitted successfully!");
        return true;
      }
    </script>
