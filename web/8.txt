    <script>
      function computeAverage() {
        var reg = document.getElementById("regno").value.trim();
        var sub = document.getElementById("subcode").value.trim();
        var t1 = document.getElementById("test1").value.trim();
        var t2 = document.getElementById("test2").value.trim();
        var t3 = document.getElementById("test3").value.trim();

        if (!reg || !sub || !t1 || !t2 || !t3) {
          alert("Please fill in all fields.");
          return false;
        }

        t1 = parseFloat(t1);
        t2 = parseFloat(t2);
        t3 = parseFloat(t3);

        if (
          isNaN(t1) ||
          isNaN(t2) ||
          isNaN(t3) ||
          t1 < 0 ||
          t1 > 100 ||
          t2 < 0 ||
          t2 > 100 ||
          t3 < 0 ||
          t3 > 100
        ) {
          alert("Enter valid marks between 0 and 100.");
          return;
        }

        var scores = [t1, t2, t3].sort();
        var avg = (scores[1] + scores[2]) / 2;
        alert("Average of Best Two Tests: " + avg.toFixed(2));
      }
    </script>
